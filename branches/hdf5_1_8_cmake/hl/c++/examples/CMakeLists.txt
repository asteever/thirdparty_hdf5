cmake_minimum_required (VERSION 2.8)
PROJECT (HDF5_HL_CPP_EXAMPLES)

#-----------------------------------------------------------------------------
# Setup include Directories
#-----------------------------------------------------------------------------
INCLUDE_DIRECTORIES (${HDF5_HL_SOURCE_DIR}/src)
INCLUDE_DIRECTORIES (${HDF5_HL_CPP_SOURCE_DIR}/src)

# --------------------------------------------------------------------
#  Add in the examples for the Packet Table codes
# --------------------------------------------------------------------
IF (HDF5_BUILD_EXAMPLES)
  ADD_EXECUTABLE (ptExampleFL ${HDF5_HL_CPP_EXAMPLES_SOURCE_DIR}/ptExampleFL.cpp)
  IF (BUILD_SHARED_LIBS)
    IF (WIN32)
      SET_PROPERTY (TARGET ptExampleFL 
          APPEND PROPERTY COMPILE_DEFINITIONS 
              _HDF5USEHLDLL_
			  _HDF5USEDLL_
      )
      SET_TARGET_PROPERTIES (ptExampleFL PROPERTIES OUTPUT_NAME "ptExampleFL")
      SET_TARGET_PROPERTIES (ptExampleFL PROPERTIES PREFIX "dll")
    ENDIF (WIN32)
  ENDIF (BUILD_SHARED_LIBS)
  TARGET_LINK_LIBRARIES (
      ptExampleFL
      ${HDF5_HL_CPP_LIB_TARGET}
      ${HDF5_HL_LIB_TARGET}
      ${HDF5_LIB_TARGET}
  )

  ADD_EXECUTABLE (ptExampleVL ${HDF5_HL_CPP_EXAMPLES_SOURCE_DIR}/ptExampleVL.cpp)
  IF (BUILD_SHARED_LIBS)
    IF (WIN32)
      SET_PROPERTY (TARGET ptExampleVL 
          APPEND PROPERTY COMPILE_DEFINITIONS 
              _HDF5USEHLDLL_
			  _HDF5USEDLL_
      )
      SET_TARGET_PROPERTIES (ptExampleVL PROPERTIES OUTPUT_NAME "ptExampleVL")
      SET_TARGET_PROPERTIES (ptExampleVL PROPERTIES PREFIX "dll")
    ENDIF (WIN32)
  ENDIF (BUILD_SHARED_LIBS)
  TARGET_LINK_LIBRARIES (
      ptExampleVL
      ${HDF5_HL_CPP_LIB_TARGET}
      ${HDF5_HL_LIB_TARGET}
      ${HDF5_LIB_TARGET}
  )
ENDIF (HDF5_BUILD_EXAMPLES)
